cmake_minimum_required(VERSION 3.16)
project(simdjson_jni LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Source directory — set by the build script when invoked via a temp-dir wrapper
set(SIMDJSON_NATIVE_DIR "${CMAKE_CURRENT_SOURCE_DIR}" CACHE PATH "Path to simdjson native sources")

include(FetchContent)
FetchContent_Declare(simdjson
    GIT_REPOSITORY https://github.com/simdjson/simdjson.git
    GIT_TAG        v4.6.1
    GIT_SHALLOW    ON
)
FetchContent_MakeAvailable(simdjson)
set_target_properties(simdjson PROPERTIES EXCLUDE_FROM_ALL TRUE)
set(SIMDJSON_SINGLEHEADER_DIR "${simdjson_SOURCE_DIR}/singleheader")

add_library(simdjson_jni SHARED
    ${SIMDJSON_NATIVE_DIR}/simdjson_c.cpp
    ${SIMDJSON_NATIVE_DIR}/simdjson_jni.cpp
    ${SIMDJSON_SINGLEHEADER_DIR}/simdjson.cpp
)

target_include_directories(simdjson_jni
    PRIVATE ${SIMDJSON_NATIVE_DIR}
    PRIVATE ${SIMDJSON_SINGLEHEADER_DIR}
)

set_target_properties(simdjson_jni PROPERTIES
    POSITION_INDEPENDENT_CODE ON
)

target_link_libraries(simdjson_jni log)
